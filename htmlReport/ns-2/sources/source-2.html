         this.currency = currency;
     }

         return amount;
     }
 
     public String getCurrency() {
         return currency;
     }

     }

     public boolean equals(Object o) {
         if (this == o) return true;
         if (o == null || getClass() != o.getClass()) return false;
         Money money = (Money) o;

     }

     }
